\uplevel{
\begin{theorem}
\label{thm:trigonometric-induction-identity}
Let $n$ be a positive integer. Then for all $\cos(x) \neq 0$ we have
\[
	1 + \sin^2(x) + \sin^4(x) + \cdots + \sin^{2n}(x) = \frac{1 - \sin^{2(n + 1)}(x)}{\cos^2(x)}.
\]
\end{theorem}
}
