\begin{parts}

\part[5] 
Prove Proposition~\ref{prop:sum-of-reciprocal-odd-products} by mathematical induction.

\begin{EnvFullwidth}
\begin{solutionorgrid}[5in]
\begin{proof}
$P(n)$ is: $\displaystyle{\sum_{j = 1}^n \frac{1}{(2j - 1)(2j + 1)} = \frac{n}{2n + 1}}$, for all $n \in \Z^+$.

\textbf{Basis}: If $n = 1$, then the LHS is
\[
	\frac{1}{(2 \times 1 - 1)(2 \times 1 + 1)} = \frac{1}{1 \times 3} = \frac{1}{3},
\]
and the RHS is
\[
	\frac{1}{2 \times 1 + 1} = \frac{1}{3}.
\]
So, $P(1)$ is true.

\textbf{Inductive step}: If $P(k)$ is true, then for all $k \in \Z^+$,
\[
	\sum_{j = 1}^k \frac{1}{(2j - 1)(2j + 1)} = \frac{k}{2k + 1}.
\]
Now,
\begin{align*}
	\sum_{j = 1}^{k + 1} \frac{1}{(2j - 1)(2j + 1)} &= \sum_{j = 1}^k \frac{1}{(2j - 1)(2j + 1)} + \frac{1}{(2(k + 1) - 1)(2(k + 1) + 1)} \\
	&= \frac{k}{2k + 1} + \frac{1}{(2k + 1)(2k + 3)} && (\textrm{by hypothesis}) \\
	&= \frac{2k^2 + 3k + 1}{(2k + 1)(2k + 3)} \\
	&= \frac{\cancel{(2k + 1)}(k + 1)}{\cancel{(2k + 1)}(2k + 3)} \\
	&= \frac{k + 1}{2(k + 1) + 1}.
\end{align*}
Since $P(1)$ is true and $P(k) \implies P(k + 1)$, by the PMI $P(n)$ is true for all $n \in \Z^+$.
\end{proof}
\end{solutionorgrid}
\end{EnvFullwidth}

\part[2]
Hence, find $\displaystyle{S = \frac{1}{3} + \frac{1}{15} + \frac{1}{35} + \cdots + \frac{1}{399}}$.

\begin{EnvFullwidth}
\begin{solutionorgrid}[2in]
As $399 = 19 \times 21$, on putting $n = 10$ into the formula in Proposition~\ref{prop:sum-of-reciprocal-odd-products} there is
\[
    S = \frac{10}{2 \times 10 + 1} = \frac{10}{21}.    
\]
\end{solutionorgrid}
\end{EnvFullwidth}
